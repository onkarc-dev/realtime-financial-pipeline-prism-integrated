cmake_minimum_required(VERSION 3.20)
project(RealtimeFinancialPipelineBenchmark LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

add_executable(benchmark_engine
    cpp_engine/benchmarks/benchmark_engine.cpp
)

if(MSVC)
    target_compile_options(benchmark_engine PRIVATE /O2 /Ob3 /DNDEBUG)
else()
    target_compile_options(benchmark_engine PRIVATE -O3 -DNDEBUG -march=native)
endif()
